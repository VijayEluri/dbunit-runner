192     private void close(Connection conn, Statement stmt, ResultSet rs) {
193         SQLException failureCause = null;
194         try {
195             if (rs != null) rs.close();
196         } catch (SQLException e) {
197             failureCause = e;
198         }
199         try {
200             if (stmt != null) stmt.close();
201         } catch (SQLException e) {
202             failureCause = failureCause == null ? e : failureCause;
203         }
204         try {
205             if (conn != null) conn.close();
206         } catch (SQLException e) {
207             failureCause = failureCause == null ? e : failureCause;
208         }
209         if (failureCause != null) throw new RuntimeException(failureCause);
210     }
